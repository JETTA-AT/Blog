Spark安装手册

Posted on 2018-12-11 |

1.Java运行环境

下载JDK

wget –no-cookies –no-check-certificate –header “Cookie: gpw_e24=http%3A%2F%2Fwww.oracle.com%2F; oraclelicense=accept-securebackup-cookie” “http://download.oracle.com/otn-pub/java/jdk/8u171-b11/512cd62ec5174c3487ac17c61aaa89e8/jdk-8u171-linux-x64.tar.gz"

tar -zxvf jdk-8u171-linux-x64.tar.gz
mv jdk1.8.0_171/ /usr/local/

vim /etc/profile
–添加以下
set java environment
exportJAVA_HOME=/usr/local/jdk1.8.0_171
exportJRE_HOME=${JAVA_HOME}/jre
exportCLASSPATH=.:${JAVA_HOME}/lib/dt.JAVA_HOME/lib/tools.jar:${JRE_HOME}/lib
exportPATH=${JAVA_HOME}/bin:${PATH}

–使用命令使环境变量立即生效
source /etc/profile

vim /etc/environment
–添加以下
PATH=”/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin:/usr/games:/usr/local/games:$JAVA_HOME/bin”
export CLASSPATH=.:$JAVA_HOME/lib:$JAVA_HOME/jre/lib
export JAVA_HOME=/usr/local/jdk1.8.0_171

–输入以下命令使环境变量立即生效
source /etc/environment

2.hadoop

hadoop版本下载

hadoop-2.7.6.tar.gz
wget “http://mirror.bit.edu.cn/apache/hadoop/common/hadoop-2.7.6/hadoop-2.7.6.tar.gz"

tar -zxvf hadoop-2.7.6.tar.gz -C /usr/

vim ~/.bashrc

环境变量配置

export HADOOP_HOME=/usr/hadoop-2.7.6
export PATH=$PATH:$HADOOP_HOME/bin:$HADOOP_HOME/sbin

vim /usr/hadoop-2.7.6/etc/hadoop/hadoop-env.sh
export JAVA_HOME=/usr/local/jdk1.8.0_171

配置文件设置

core-site.xml

vim /usr/hadoop-2.7.6/etc/hadoop/core-site.xml


fs.defaultFS
/data


hadoop.tmp.dir
/root/hadoop/tmp

vim /usr/hadoop-2.7.6/etc/hadoop/core-site.xml


fs.default.name
hdfs://localhost:9000

hdfs-site.xml

vim /usr/hadoop-2.7.6/etc/hadoop/hdfs-site.xml



dfs.replication
1



dfs.name.dir
/data/hdfs/namenode



dfs.data.dir
/data/hdfs/datanode

yarn-site.xml

vim /usr/hadoop-2.7.6/etc/hadoop/yarn-site.xml



yarn.nodemanager.aux-services
mapreduce_shuffle

hadoop初始化和启动

初始化hadoop：
/usr/hadoop-2.7.6/bin/hdfs namenode -format

启动
$HADOOP_HOME/sbin/start-all.sh

scala安装

wget “https://downloads.lightbend.com/scala/2.11.12/scala-2.11.12.tgz"

tar -zxvf scala-2.11.12.tgz
mv scala-2.11.12/ /usr/local/

vim /etc/profile
export PATH=”$PATH:/usr/local/scala-2.11.12/bin”

spark安装

wget “http://mirror.bit.edu.cn/apache/spark/spark-2.3.0/spark-2.3.0-bin-hadoop2.7.tgz"

tar -zxvf spark-2.3.0-bin-hadoop2.7.tgz -C /usr/

–添加环境变量
vim ~/.bashrc

export SPARK_HOME=/usr/spark-2.3.0-bin-hadoop2.7
export PATH=$SPARK_HOME/bin:$SPARK_HOME/sbin:$PATH

cp /usr/spark-2.3.0-bin-hadoop2.7/conf/spark-env.sh.template /usr/spark-2.3.0-bin-hadoop2.7/conf/spark-env.sh

vim /usr/spark-2.3.0-bin-hadoop2.7/conf/spark-env.sh

export SPARK_MASTER_IP=localhost
export SPARK_WORKER_MEMORY=8g
export JAVA_HOME=/usr/local/jdk1.8.0_171
export SCALA_HOME=/usr/local/scala-2.11.12
export SPARK_HOME=/usr/spark-2.3.0-bin-hadoop2.7
export HADOOP_CONF_DIR=/usr/hadoop-2.7.6/etc/hadoop
export SPARK_LIBRARY_PATH=$SPARK_HOME/lib
export SCALA_LIBRARY_PATH=$SPARK_LIBRARY_PATH
export SPARK_WORKER_CORES=5
export SPARK_WORKER_INSTANCES=1
export SPARK_MASTER_PORT=7077
